# regular empires
inline_script = {
    script = generic_parts/fwt_toggled_job
    job = artisan
    add = $add$
    type = $type$
    count = $jobs$
    condition = "
        owner = {
            is_gestalt = no
        }
        NOT = {	has_modifier = penal_colony }
        $condition$
    "
    zone = "_district"
    multiplier_string = "$multiplier_string$"
}
inline_script = {
    script = generic_parts/giga_toggled_desc
    toggle = @[ $jobs$ * $description$ ]
    trigger = "
        owner = {
            is_gestalt = no
            artisan_is_artificer_trigger = no
            artisan_is_pearl_diver_trigger = no
        }
        NOT = {	has_modifier = penal_colony }
        $condition$
    "
    text = job_artisan_effect_desc
}
