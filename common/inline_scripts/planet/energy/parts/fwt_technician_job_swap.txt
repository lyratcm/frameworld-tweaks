# regular empires
inline_script = {
    script = generic_parts/fwt_toggled_job
    job = technician
    add = $add$
    type = $type$
    count = $jobs$
    condition = "
        owner = {
            is_gestalt = no
            is_fallen_empire_spiritualist = no
        }
        $condition$
    "
    zone = "_district"
    multiplier_string = "$multiplier_string$"
}
inline_script = {
    script = generic_parts/giga_toggled_desc
    toggle = @[ $jobs$ * $description$ ]
    trigger = "
        owner = {
            is_gestalt = no
            is_fallen_empire_spiritualist = no
        }
        $condition$
    "
    text = job_technician_effect_desc
}
