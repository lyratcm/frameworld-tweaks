# regular empires
inline_script = {
    script = generic_parts/giga_toggled_job
    job = soldier
    add = $add$
    type = $type$
    count = $jobs$
    condition = "
        owner = {
            is_gestalt = no
        }
        $condition$
    "
    multiplier_string = "$multiplier_string$"
}
inline_script = {
    script = generic_parts/giga_toggled_desc
    toggle = @[ $jobs$ * $description$ ]
    trigger = "
        owner = {
            is_gestalt = no
        }
        soldier_is_necromancer = no
        $condition$
    "
    text = job_soldier_effect_desc
}
inline_script = {
    script = generic_parts/giga_toggled_desc
    toggle = @[ $jobs$ * $description$ ]
    trigger = "
        owner = {
            is_gestalt = no
        }
        soldier_is_necromancer = yes
        $condition$
    "
    text = job_necromancer_effect_desc
}
