# regular empires
inline_script = {
    script = generic_parts/giga_toggled_job
    job = enforcer
    add = $add$
    type = $type$
    count = $jobs$
    condition = "
        owner = {
            is_gestalt = no
        }
        $condition$
    "
    multiplier_string = "$multiplier_string$"
}
inline_script = {
    script = generic_parts/giga_toggled_desc
    toggle = @[ $jobs$ * $description$ ]
    trigger = "
        owner = {
            is_gestalt = no
        }
        enforcer_is_educator_trigger = no
        enforcer_is_telepath_trigger = no
        $condition$
    "
    text = job_enforcer_effect_desc
}
