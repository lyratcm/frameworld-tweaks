
# vanilla

# merchants
inline_script = {
    script = generic_parts/fwt_toggled_job
    job = merchant
    add = $add$
    count = $jobs$
    condition = "
        owner = {
            has_civic = civic_merchant_guilds
        }
        $condition$
    "
    zone = "_district"
    description = $description$
    multiplier_string = "$multiplier_string$"
    type = "$type$"
}
