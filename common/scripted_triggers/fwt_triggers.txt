fwt_tech_trigger_not = {
    owner = {
        not = {
            has_technology = $TECHNOLOGY$
        }
    }
}

fwt_tech_trigger = {
    owner = {
        has_technology = $TECHNOLOGY$
    }
}

    }

is_asteroid_harvestable = {
    [[PLANET_1]
        solar_system = {
            count_system_planet = {
                count > 1
                limit = {
                    is_asteroid = yes
                }
            }
        }
    ]
    [[PLANET_STRATEGIC]
        or = {
            planet_resource_compare = {
                type = produces
                resource = rare_crystals
                value > 0
            }
            planet_resource_compare = {
                type = produces
                resource = exotic_gases
                value > 0
            }
            planet_resource_compare = {
                type = produces
                resource = volatile_motes
                value > 0
            }
        }
    ]
    [[PLANET_INDUSTRY]
        is_asteroid = yes
        is_moon = no
        has_anomaly = no
        planet_has_no_megastructure = yes
    ]
}
